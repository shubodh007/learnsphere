## 🚀 Quick Start (All Options)

### 1. Basic Chat (Option 1)
```tsx
import { MarkdownRenderer } from '@/components/MarkdownRenderer';
import { LoadingMessages } from '@/components/LoadingMessages';

// In your chat
{isLoading && <LoadingMessages />}
{message.content && <MarkdownRenderer content={message.content} />}
```

### 2. Educational Mode (Option 2)
```tsx
import { chatConfig } from '@/config/chat-config';
import { EducationalLoadingMessages } from '@/components/EducationalLoadingMessages';
import { DifficultyBadge, CalloutBox } from '@/components/EducationalFormatter';

// Switch to educational loading
chatConfig.loading.style = 'educational';

// Use educational components
<DifficultyBadge level="intermediate" />
<CalloutBox type="tip">Remember to practice!</CalloutBox>
```

### 3. Interactive Learning (Option 3)
```tsx
import { ChatMessage } from '@/components/ChatMessage';
import { QuizComponent } from '@/components/QuizComponent';
import { ConceptTemplate } from '@/components/ResponseTemplates';

// Enhanced messages
<ChatMessage
  id={msg.id}
  role={msg.role}
  content={msg.content}
  difficulty="intermediate"
  onFeedback={handleFeedback}
/>

// Interactive quiz
<QuizComponent question={quizData} />

// Professional template
<ConceptTemplate title="Hooks" {...data} />
```
